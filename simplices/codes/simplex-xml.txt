simplex <- function(P,filepath,title) {


	A=1:P
	
	SET <- array(c(0),dim= c((P),P))	
	for (i in A){
		SET[i,i] <- 1
		}
	Final <- f.composition(SET)

	A <- nrow(Final)
	F <- NULL
	for ( i in 1:(A-1)){
		B=1:A
		B <- B[-(1:i)]
		for (j in B){
			Full.wires <- matrix(rbind(F,c(i,j)),ncol=2)
			F <- Full.wires
			}
		}

	f.write.xml(Final,Full,filepath,title)
}
